# Use the official .NET runtime as the base image
FROM mcr.microsoft.com/dotnet/runtime:7.0 AS base

# Set the working directory in the container
WORKDIR /app

# Copy the BusyFileWriter.exe and any required files to the container
COPY BusyFileWriter.exe ./

# Specify a mount point for the log output
VOLUME /log

# Define the command to run your console app
ENTRYPOINT ["dotnet", "BusyFileWriter.exe"]
